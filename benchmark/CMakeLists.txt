function(CURVE_BENCHMARK_SET tb tf tc nb base nbt m8 mt ct pf)
    set(bd "${tb}_${base}")
    file(READ "benchtest_ec_ZZZ.c" tempc)
    string(REGEX REPLACE "XXX" ${bd} tempc "${tempc}")
    string(REGEX REPLACE "ZZZ" ${tc} tempc "${tempc}")
    file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/benchtest_ec_${tc}.c" "${tempc}")
    add_executable (benchtest_ec_${tc} ${CMAKE_CURRENT_BINARY_DIR}/benchtest_ec_${tc}.c)
    target_link_libraries (benchtest_ec_${tc} amcl_curve_${tc})

  if(NOT(${pf} STREQUAL "NOT"))
    file(READ "benchtest_pair_ZZZ.c" tempc)
    string(REGEX REPLACE "XXX" ${bd} tempc "${tempc}")
    string(REGEX REPLACE "YYY" ${tf} tempc "${tempc}")
    string(REGEX REPLACE "ZZZ" ${tc} tempc "${tempc}")
    file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/benchtest_pair_${tc}.c" "${tempc}")
    add_executable (benchtest_pair_${tc} ${CMAKE_CURRENT_BINARY_DIR}/benchtest_pair_${tc}.c)
    target_link_libraries (benchtest_pair_${tc} amcl_pairing_${tc})
  endif(NOT(${pf} STREQUAL "NOT"))
endfunction(CURVE_BENCHMARK_SET tb tf tc nb base nbt m8 mt ct pf)
